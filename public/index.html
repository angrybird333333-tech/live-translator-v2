const express = require('express'); const line = require('@line/bot-sdk'); const axios = require('axios'); const lineConfig = { channelSecret: process.env.LINE_CHANNEL_SECRET, channelAccessToken: process.env.LINE_CHANNEL_ACCESS_TOKEN, }; const lineClient = new line.messagingApi.MessagingApiClient({ channelAccessToken: process.env.LINE_CHANNEL_ACCESS_TOKEN, }); const app = express(); app.get('/ping', (req, res) => res.send('pong')); app.post('/webhook', line.middleware(lineConfig), async (req, res) => { res.sendStatus(200); const events = req.body.events; await Promise.all(events.map(handleEvent)); }); async function handleEvent(event) { if (event.type !== 'message' || event.message.type !== 'text') return; const userText = event.message.text; const isThai = /[\u0E00-\u0E7F]/.test(userText); const sourceLang = isThai ? 'ไทย' : '中文'; const targetLang = isThai ? '繁體中文' : 'ภาษาไทย'; try { const response = await axios.post( 'https://api.anthropic.com/v1/messages', { model: 'claude-haiku-4-5', max_tokens: 1024, messages: [ { role: 'user', content: `請將以下${sourceLang}翻譯成${targetLang}，只輸出翻譯結果，不要加任何說明或解釋：\n${userText}` } ] }, { headers: { 'x-api-key': process.env.ANTHROPIC_API_KEY, 'anthropic-version': '2023-06-01', 'content-type': 'application/json' } } ); const translated = response.data.content[0].text.trim(); return lineClient.replyMessage({ replyToken: event.replyToken, messages: [{ type: 'text', text: `${userText}\n\n📝 ${translated}` }], }); } catch (error) { console.error('翻譯錯誤:', error.message); return lineClient.replyMessage({ replyToken: event.replyToken, messages: [{ type: 'text', text: '翻譯暫時無法使用，請稍後再試 🙏' }], }); } } const PORT = process.env.PORT || 3000; app.listen(PORT, '0.0.0.0', () => { console.log('Bot running on port', PORT); });
